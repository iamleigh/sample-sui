    </PropSection>

    </PropSection>

	<PropSection title="isResponsive" type="boolean" defaultValue="false">

    </PropSection>

    </PropSection>

	<PropSection title="isDisabled" type="boolean">

    </PropSection>

	<PropSection title="isAutoClose" type="boolean" defaultValue="true">
