<PackageInstallGuide packageName="@wpmudev/sui-checkbox" />

<Section title="Checkbox Props" border={true}>
	Use the <Code>Checkbox</Code> component to create a checkbox input. It can be used as a standalone checkbox or as part of a group of checkboxes. It has the following props:

	<PropSection title="id" type="string">
    	A unique identifier for the checkbox, within group context this will be generated automatically.
    </PropSection>

		<PropSection title="name" type="string">
    		Represents the name of the input checkbox.
    	</PropSection>

		<PropSection title="value" type="string">
    		Represents the value of the input checkbox.
    	</PropSection>

    	<PropSection title="label" type="string">
    		The text label for the checkbox.
    	</PropSection>

		<PropSection title="isLabelHidden" type="boolean" defaultValue="false">
    		Indicates whether the label should be hidden or not.
    	</PropSection>

		<PropSection title="isSmall" type="boolean" defaultValue="false">
    		if set to <Code isSmall={true}>true</Code> the checkbox will have a smaller size.
    	</PropSection>

		<PropSection title="isDisabled" type="boolean" defaultValue="false">
    		Indicates whether the checkbox is disabled or not.
    	</PropSection>

		<PropSection title="isError" type="boolean" defaultValue="false">
    		Indicates if the checkbox has an error.
    	</PropSection>

		<PropSection title="isIndeterminate" type="boolean" defaultValue="false">
    		Mark checkbox checked as indeterminate, for more information checkout the <a target="_blank" href="/?path=/docs/sui-components-forms-checkbox--docs">Usage Page</a>
    	</PropSection>

		<PropSection title="isChecked" type="boolean" defaultValue="false">
    		Default value for the checkbox, when set to <Code small={true}>true</Code> the Checkbox will be checked initially.
    	</PropSection>

		<PropSection title="onChange" type="function">
    		A callback function to be invoked when the checkbox state is changed.
    	</PropSection>

		<FormValidationProps />
</Section>
